```typescript
// Recover orphaned dispatch records (CRASH-01/02).
// At startup there are zero running agents, so any record in an active
// (in-flight) stage is definitionally orphaned — its agent crashed.
// Transition each to failed and reopen the issue so it re-enters the queue.
// Records with stage=implemented are NOT touched — they survive the crash
// and will be picked up for Sentinel dispatch on the next tick (CRASH-03).
const orphanedStages: import("./types.js").DispatchStage[] = [
  "scouting",
  "implementing",
  "reviewing",
];
for (const rec of store.all()) {
  if (orphanedStages.includes(rec.stage)) {
    try {
      store.transition(rec.issue_id, "failed", { current_agent_id: null });
      await beads.reopen(rec.issue_id);
      this.emit({
        type: "orchestrator.recovered_record",
        data: { issue_id: rec.issue_id, was_stage: rec.stage },
        timestamp: Date.now(),
      });
    } catch {
      // best-effort — continue with other records
    }
  }
}
```
