**New dispatch rules (stage switch, in order):**
```typescript
const stage = record?.stage ?? "pending";

switch (stage) {
  case "pending":
    if (countByCaste(runningAgents, "oracle") >= concurrencyLimits.max_oracles) {
      return { type: "skip", issue, reason: "concurrency limit" };
    }
    return { type: "dispatch_oracle", issue };

  case "scouted":
    if (issue.status !== "open" && issue.status !== "ready") {
      return { type: "skip", issue, reason: "issue not ready" };
    }
    if (countByCaste(runningAgents, "titan") >= concurrencyLimits.max_titans) {
      return { type: "skip", issue, reason: "concurrency limit" };
    }
    return { type: "dispatch_titan", issue };

  case "implemented":
    if (countByCaste(runningAgents, "sentinel") >= concurrencyLimits.max_sentinels) {
      return { type: "skip", issue, reason: "concurrency limit" };
    }
    return { type: "dispatch_sentinel", issue };

  case "scouting":
  case "implementing":
  case "reviewing":
    return { type: "skip", issue, reason: "agent running" };

  case "complete":
    return { type: "skip", issue, reason: "complete" };

  case "failed":
    return { type: "skip", issue, reason: "failed" };

  default: {
    const _exhaustive: never = stage;
    return { type: "skip", issue, reason: `unknown stage: ${_exhaustive}` };
  }
}
```
